10001st prime

Problem 7

By listing the first six prime numbers: 2, 3, 5, 7, 11, and 13, we can see that the 6th prime is 13.

What is the 10 001st prime number?

前六个质数为：2, 3, 5, 7, 11, 13。我们可以看到，第六个质数为13。

那么第10001个质数是多少？

Solution 1


function isPrime(num){ //判断是否为素数
	for(var i=2; i<=Math.ceil(num/2);i++){
		if(num%i == 0){
			return false;
		}
	}
	return true;
}

var count = 0
for(var i=2; ; i++){
	if(isPrime(i)){
		count++;
	}
	if(count == 10001){
		console.log(i);
		break;
	}
}

/*质数的一些性质：
- 1不是质数
- 除了2的所有质数都是奇数
- 如果找不到这样一个数字f，这个f是小于等于n的平方根，但是又可以被n整除的，那么n就是质数。
- 所有大于3的质数都可以写成 6k+/-1的形式
	证明：
	P>3且是质数
	故，P为奇数，且无3的因数
	1) 若P=3t+1，且 p=2s+1
	则p=2s+1=3t+1
	p-1=2s=3t
	则p-1=6k,p=6k+1
	2) 若P=3t+2，且 p=2s+1
	则p=2s+1=3t+2
	p+1=2(s+1)=3(t+1)
	故p+1=6k
	p=6k-1*/

function isPrime(n){
	if(n==1){
		return false;
	} else if(n<4){
		return true; //2和3是质数
	} else if(n%2==0){

		return true; //因为4、6、8已经被剔除了
	} else if(n%3==0){
		return false;
	} else {
		var r = Math.floor(Math.pow(n, 0.5)); //r是小于等于n的平方根的最大数
		var f = 5;

			}
			if(n%(f+2)==0){ //这里测6k+1

			}

		}

	}
			
}
