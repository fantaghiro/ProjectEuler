Multiples of 3 and 5

Problem 1

If we list all the natural numbers below 10 that are multiples of 3 or 5, we get 3, 5, 6 and 9. The sum of these multiples is 23.

Find the sum of all the multiples of 3 or 5 below 1000.

如果我们把10以下可以被3或5整除的数列出来，我们会得到3、5、6、9，这几个数相加，和为23。

请求出1000以下可以被3或5整除的所有数字之和。

Solution 1


function multiplesOf3and5(num){
	var sum = 0;
	for(var i=0; i<num; i++){
		if(i%3==0 || i%5==0){
			sum += i;
		}
	}
	return sum;
}

alert(multiplesOf3and5(1000));

Solution 2


function sumDivisible(div, max){
	//计算max以内(max含)能够被div整除的所有整数的和
	
	//根据等差序列求和公式，1到n的数字之和为 1/2*(1+n)*n
	
	//11及以下能够被2整除：2+4+6+8+10 = 2*(1+2+3+4+5) = 2 * [1/2 * (1+5) * 5]
	//注意Math.floor(11/2)也是5
	
	var n = Math.floor(max/div);		
	return div * 1/2 * (1 + n) * n;
}

//计算1000以内（1000不含）中能被3或5整除的数字的和
var result = sumDivisible(3, 999) + sumDivisible(5, 999) - sumDivisible(3*5, 999);
