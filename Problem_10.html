Summation of primes

Problem 10

The sum of the primes below 10 is 2 + 3 + 5 + 7 = 17.

Find the sum of all the primes below two million.

小于10的所有质数相加为：2 + 3 + 5 + 7 = 17。

求小于2,000,000的所有质数之和。

Solution 1


var n = 5; //已知2、3为质数
var sum = 5; //已知2、3的和为5

//这种判断质数的方法，详见Problem 7 Solution 2的解释
function isPrime(n){
	if(n==1){
		return false;
	} else if(n<4){
		return true; //2和3是质数
	} else if(n%2==0){
		return false;
	} else if (n<9) {
		return true; //因为4、6、8已经被剔除了
	} else if(n%3==0){
		return false;
	} else {
		var r = Math.floor(Math.pow(n, 0.5)); //r是小于等于n的平方根的最大数
		var f = 5;
		while(f <= r){
			if(n%f==0){ //这里是测6k-1
				return false;
			}
			if(n%(f+2)==0){ //这里测6k+1
				return false;
			}
			f = f + 6;
		}
		return true;
	}			
}
while(n < 2000000 ){
	if(isPrime(n)){
		sum += n;
	}
	n+=2;
	if((n<2000000)&&(isPrime(n))){
		sum += n;
	}
	n+=4;
}

console.log(sum);
