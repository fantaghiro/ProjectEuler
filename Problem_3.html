Largest prime factor

Problem 3

The prime factors of 13195 are 5, 7, 13 and 29.

What is the largest prime factor of the number 600851475143 ?

13195的素数因子是5、7、13和29。

数字600851475143的最大素数因子为多少？

Solution 1


var result = 600851475143;
var arr = [];

function isPrime(num){ //判断是否为素数
	for(var i=2; i<=Math.ceil(num/2);i++){
		if(num%i == 0){
			return false;
		}
	}
	return true;
}

for(var i=2; i<=result; i++){
	if((result%i == 0) && isPrime(i)){
		arr.push(i);
		result = result / i;
	}
}

arr.sort(function(a,b){return b-a});
console.log(arr[0]);

Solution 2


var result = 600851475143;
var factor = 2;
var largestFactor = 1;
while(result > 1){ //从小的因子开始除，一直除到不能再除，因子加1，然后继续检测
	if(result%factor == 0){
		largestFactor = factor;
		result = result / factor;
		while(result%factor == 0){
			result = result / factor;
		}
	}
	factor = factor + 1;
}
console.log(largestFactor);

var largestFactor = 1;
var factor = 3;

	result = result/2;
	largestFactor = 2;
	while(result%2 == 0){
		result = result/2;
	}
}

while(result > 1){

		}
	}
	factor = factor + 2; //这里两个两个的加
}

console.log(largestFactor);

var largestFactor = 1;

var maxFactor = 1; //一个数至多只能有一个大于其平方根的最大因子

//由于2是最小的偶数素数，因此完全可以让因子两个两个的加，但是首先要把2都先除尽
if(result%2 == 0){
	result = result/2;
	largestFactor = 2;
	while(result%2 == 0){
		result = result/2;
	}
}
maxFactor = Math.sqrt(result);
while(result>1 && largestFactor<=maxFactor){

		result = result / factor;

			result = result / factor;
		}
		maxFactor = Math.sqrt(result);
	}

}
